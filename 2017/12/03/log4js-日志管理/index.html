log4js 日志管理

2017-12-03

使用方法

log4js.json配置方法

log4js.configure({

appenders: {

    ruleConsole: {type: 'console'},

    ruleFile: {

        type: 'dateFile',

        filename: 'logs/server-',

        pattern: 'yyyy-MM-dd.log',

        maxLogSize: 10 * 1000 * 1000,

        numBackups: 3,

        alwaysIncludePattern: true

    }

},

categories: {

    default: {appenders: ['ruleConsole', 'ruleFile'], level: 'info'}

}

});

```
